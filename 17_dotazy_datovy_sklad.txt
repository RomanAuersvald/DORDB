--Dotaz pro získání průměrné fakturované částky pro uživatele za předešlé fiskální roky. - otestovat
SELECT dwuser.dw_userid, concat(dwuser.firstname, dwuser.lastname) as jmeno, avg(dwinvoices.totalprice) AS AVGAmount, dwfiscalyear.dw_fiscalyearid
FROM DWUser inner join DWInvoices on DWInvoices.dw_userId  = DWUser.dw_userId inner join DWFiscalYear on DWFiscalYear.dw_fiscalYearId = DWInvoices.dw_fiscalYearId inner join DWInvoice on DWInvoices.dw_invoiceId = DWInvoice.dw_invoiceId
GROUP BY dwuser.dw_userid, DWFiscalYear.dw_fiscalYearId, concat(dwuser.firstname, dwuser.lastname);


--Dotaz pro získání 3 nejčastějších klientů pro uživatele za poslední fiskální rok. - otestovat
select concat(dwuser.firstname, dwuser.lastname) as jmeno, DWClient.companyname as Client, dwuser.dw_userid, count(dwclient.dw_clientId) as Contracts, max(dwfiscalyear.dw_fiscalyearId) as lastYearID
from DWInvoices inner join dwuser on dwuser.dw_userId = DWInvoices.dw_userId inner join DWClient on DWInvoices.dw_clientId = dwclient.dw_clientId inner join dwfiscalyear on dwfiscalyear.dw_fiscalyearId = dwinvoices.dw_fiscalyearId
where dwfiscalyear.dw_fiscalyearId = 4 and rownum <= 4
GROUP BY dwuser.dw_userid, DWClient.companyname, concat(dwuser.firstname, dwuser.lastname) order by count(dwclient.dw_clientId);
